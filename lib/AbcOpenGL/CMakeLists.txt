    IPointsDrw.h

    ISubDDrw.h
    IXformDrw.h
    MeshDrwHelper.h

    ICurvesDrw.cpp
    INuPatchDrw.cpp
    IObjectDrw.cpp
    IPointsDrw.cpp
    IPolyMeshDrw.cpp
    ISubDDrw.cpp

        DESTINATION include/AbcOpenGL
        PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
